## 📍 PIN diagram of bgos v2.0 Project

### ⬇️ BUTTON PINS and BUZZER PIN

+ btnRight = 38

+ btnLeft = 35

+ btnUp = 37

+ btnDown = 36

+ btnSelect = 39

+ BUZZER_PIN = 15

+ macro3 = 2

+ macro4 = 1

+ hPin = 40

+ ePin = 41

+ PeakBtn = 10

+ NorthEast = 17

### ⬇️ 0.96" OLED Screen PINS (U8g2 Library in Arduino)

+ SCL = 4
  
+ SDA = 5

+ VCC = 3V3

+ GND = GND
